if (!window.onBromoLoad) { function OnBromoWindowLoad() { this.spool = new Array(); this.add = function(script) { this.spool.push(script); } this.execute = function() { while (this.spool.length > 0) { this.spool[0](); this.spool.shift(); } } } window.onBromoLoad = new OnBromoWindowLoad(); if (window.addEventListener) { window.addEventListener("load", function(){window.onBromoLoad.execute()}, false); } else if (window.attachEvent) { window.attachEvent("onload", function(){window.onBromoLoad.execute()}); } }
